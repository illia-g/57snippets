#
# Хоткеи для IDE Netbeans
#
#

# Создать в коде закладку для быстрого перехода
Ctrl + Shift + M

# Переключиться на предыдущую закладку
Ctrl + Shift + ,

# Переключиться на следующую закладку
Ctrl + Shift + .

# Перейти в место последнего изменения
Ctrl + Q

# Назад/Вперед
Alt + Left, Alt + Right

# Перейкти на вкладку проектов
Ctrl + 1

# Перейти в окно редактора
Ctrl + 0

# Закрыть текущие файл в редакторе
Ctrl + W
